# Crontab Entry for Log Cleanup Script
# This entry runs the log cleanup script daily at 12:05 AM UTC
# 
# To install this crontab entry, run:
# crontab -e
# 
# Then add this line to your crontab:

5 0 * * * /bin/bash -c "cd /home/ubuntu/TelegramScraper/telegram-ai-scraper && source telegram-ai-scraper_env/bin/activate && /home/ubuntu/TelegramScraper/telegram-ai-scraper/logs/cleanup_log_entries.sh" >> /home/ubuntu/TelegramScraper/telegram-ai-scraper/logs/cron_cleanup.log 2>&1

# Explanation:
# 5 0 * * *     = Every day at 12:05 AM UTC (5 minutes past midnight)
# /bin/bash -c  = Run the command in bash shell
# cd ...        = Change to project directory
# source ...    = Activate the Python virtual environment
# cleanup...sh  = Run the cleanup script
# >> cron...log = Append output to cron log file
# 2>&1          = Redirect stderr to stdout so errors are also logged

# Alternative version without virtual environment activation (if not needed):
# 5 0 * * * /home/ubuntu/TelegramScraper/telegram-ai-scraper/logs/cleanup_log_entries.sh >> /home/ubuntu/TelegramScraper/telegram-ai-scraper/logs/cron_cleanup.log 2>&1

# To verify crontab is working, check:
# tail -f /home/ubuntu/TelegramScraper/telegram-ai-scraper/logs/cron_cleanup.log

# To list current crontabs:
# crontab -l

# To remove a crontab:
# crontab -r
